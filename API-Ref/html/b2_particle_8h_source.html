185  static b2Inline void Mix(b2ParticleColor * const colorA,
186  b2ParticleColor * const colorB,
187  const int32 strength)
188  {
189  const uint8 dr = (strength * (colorB->r - colorA->r)) >>
190  k_bitsPerComponent;
191  const uint8 dg = (strength * (colorB->g - colorA->g)) >>
192  k_bitsPerComponent;
193  const uint8 db = (strength * (colorB->b - colorA->b)) >>
194  k_bitsPerComponent;
195  const uint8 da = (strength * (colorB->a - colorA->a)) >>
196  k_bitsPerComponent;
197  colorA->r += dr;
198  colorA->g += dg;
199  colorA->b += db;
200  colorA->a += da;
201  colorB->r -= dr;
202  colorB->g -= dg;
203  colorB->b -= db;
204  colorB->a -= da;
205  }
