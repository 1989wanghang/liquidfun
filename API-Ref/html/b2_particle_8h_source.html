200  static b2Inline void MixColors(b2ParticleColor * const colorA,
201  b2ParticleColor * const colorB,
202  const int32 strength)
203  {
204  const uint8 dr = (uint8)((strength * (colorB->r - colorA->r)) >>
205  k_bitsPerComponent);
206  const uint8 dg = (uint8)((strength * (colorB->g - colorA->g)) >>
207  k_bitsPerComponent);
208  const uint8 db = (uint8)((strength * (colorB->b - colorA->b)) >>
209  k_bitsPerComponent);
210  const uint8 da = (uint8)((strength * (colorB->a - colorA->a)) >>
211  k_bitsPerComponent);
212  colorA->r += dr;
213  colorA->g += dg;
214  colorA->b += db;
215  colorA->a += da;
216  colorB->r -= dr;
217  colorB->g -= dg;
218  colorB->b -= db;
219  colorB->a -= da;
220  }
